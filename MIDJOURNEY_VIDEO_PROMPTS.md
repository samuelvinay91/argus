```typescript
    </div>

```

### Lazy Loading for Performance:

```typescript
'use client';

import { useEffect, useRef, useState } from 'react';

export function LazyVideo({ src }: { src: string }) {
  const videoRef = useRef<HTMLVideoElement>(null);
  const [isInView, setIsInView] = useState(false);

  useEffect(() => {
    const observer = new IntersectionObserver(
      ([entry]) => {
        if (entry.isIntersecting) {
          setIsInView(true);
        }
      },
      { threshold: 0.1 }
    );

    if (videoRef.current) {
      observer.observe(videoRef.current);
    }

    return () => observer.disconnect();
  }, []);

  return (
    <video
      ref={videoRef}
      autoPlay={isInView}
      muted
      loop
      playsInline
      className="w-full h-full object-cover"
    >
      {isInView && <source src={src} type="video/mp4" />}
    </video>
  );
}
```

### 1. Compress Videos
```bash
# Using ffmpeg
ffmpeg -i input.mp4 -vcodec h264 -crf 28 -preset slow output.mp4

# Or use online tool:
# https://www.freeconvert.com/video-compressor
```

### 2. Use Poster Image
Always provide poster for instant visual:
```typescript
<video poster="/hero-poster.jpg" ...>
```

### 3. Lazy Load Below Fold
Only load hero video immediately, lazy load others

### 4. Provide WebM Alternative
Better compression for modern browsers:
```bash
ffmpeg -i input.mp4 -c:v libvpx-vp9 -crf 30 output.webm
```
